# TAXONE - Criar Views para fixar o profile do uso do dicionário core do Oracle

## Descricao

Criar Views para fixar o profile do uso do dicionário core do Oracle
 
 Essas views devem ser objetos que fazem parte do produto 
 select 'CREATE OR REPLACE FORCE VIEW '||username||'.ALL_ARGUMENTS AS 
select /*+optimizer_features_enable(''10.2.0.4'')*/ * from sys.all_arguments where owner='''||username||''';
 
 CREATE OR REPLACE FORCE VIEW '||username||'.ALL_TYPES AS 
 select /*+optimizer_features_enable(''10.2.0.4'')*/ * from sys.all_TYPES where owner='''||username||''';'
 from dba_users where username like 'USER_TAXONE_%';
  
 Exemplo: 
 CREATE OR REPLACE FORCE VIEW USER_TAXONE_CIBRAFERTIL.ALL_ARGUMENTS
 (OWNER, OBJECT_NAME, PACKAGE_NAME, OBJECT_ID, OVERLOAD, 
 SUBPROGRAM_ID, ARGUMENT_NAME, POSITION, SEQUENCE, DATA_LEVEL, 
 DATA_TYPE, DEFAULTED, DEFAULT_VALUE, DEFAULT_LENGTH, IN_OUT, 
 DATA_LENGTH, DATA_PRECISION, DATA_SCALE, RADIX, CHARACTER_SET_NAME, 
 TYPE_OWNER, TYPE_NAME, TYPE_SUBNAME, TYPE_LINK, TYPE_OBJECT_TYPE, 
 PLS_TYPE, CHAR_LENGTH, CHAR_USED, ORIGIN_CON_ID)
 BEQUEATH DEFINER AS select /*+optimizer_features_enable('10.2.0.4')*/ "OWNER","OBJECT_NAME","PACKAGE_NAME","OBJECT_ID","OVERLOAD","SUBPROGRAM_ID","ARGUMENT_NAME","POSITION","SEQUENCE","DATA_LEVEL","DATA_TYPE","DEFAULTED","DEFAULT_VALUE","DEFAULT_LENGTH","IN_OUT","DATA_LENGTH","DATA_PRECISION","DATA_SCALE","RADIX","CHARACTER_SET_NAME","TYPE_OWNER","TYPE_NAME","TYPE_SUBNAME","TYPE_LINK","TYPE_OBJECT_TYPE","PLS_TYPE","CHAR_LENGTH","CHAR_USED","ORIGIN_CON_ID" from sys.all_arguments where owner='USER_TAXONE_CIBRAFERTIL';
  
 CREATE OR REPLACE FORCE VIEW USER_TAXONE_CIBRAFERTIL.ALL_TYPES
 (OWNER, TYPE_NAME, TYPE_OID, TYPECODE, ATTRIBUTES, 
 METHODS, PREDEFINED, INCOMPLETE, FINAL, INSTANTIABLE, 
 PERSISTABLE, SUPERTYPE_OWNER, SUPERTYPE_NAME, LOCAL_ATTRIBUTES, LOCAL_METHODS, TYPEID)
 BEQUEATH DEFINER
